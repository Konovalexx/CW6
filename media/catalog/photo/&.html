def send_mailing():
    scheduler = BackgroundScheduler()

def send_mailing_job():
    zone = pytz.timezone(settings.TIME_ZONE)
    current_datetime = timezone.now()

    mailings = Mailing.objects.filter(start_time__lte=current_datetime, status='running')

    for mailing in mailings:
        last_attempt = MailingAttempt.objects.filter(mailing=mailing).order_by('-timestamp').first()
        if last_attempt and not check_frequency(last_attempt, mailing.frequency):
            continue

        message_body = f"Привет!\n\n{mailing.message.body}"

        try:
            send_mail(
                subject=mailing.message.subject,
                message=message_body,
                from_email=settings.EMAIL_HOST_USER,
                recipient_list=[client.email for client in mailing.clients.all()],
                fail_silently=False,
            )
            MailingAttempt.objects.create(mailing=mailing, status='successful', owner=mailing.owner)  # Сохраняем владельца
        except smtplib.SMTPException as e:
            MailingAttempt.objects.create(mailing=mailing, status='failed', response=str(e), owner=mailing.owner)  # Сохраняем владельца

def check_frequency(last_attempt, frequency):
    interval_mapping = {
        'daily': timezone.timedelta(days=1),
        'weekly': timezone.timedelta(weeks=1),
        'monthly': timezone.timedelta(weeks=4),
    }
    return timezone.now() - last_attempt.timestamp >= interval_mapping.get(frequency, timezone.timedelta(days=1))

def send_mailing_job():
    zone = pytz.timezone(settings.TIME_ZONE)
    current_datetime = timezone.now()

    mailings = Mailing.objects.filter(start_time__lte=current_datetime, status='running')

    for mailing in mailings:
        last_attempt = MailingAttempt.objects.filter(mailing=mailing).order_by('-timestamp').first()
        if last_attempt and not check_frequency(last_attempt, mailing.frequency):
            continue

        message_body = f"Привет!\n\n{mailing.message.body}"

        try:
            send_mail(
                subject=mailing.message.subject,
                message=message_body,
                from_email=settings.EMAIL_HOST_USER,
                recipient_list=[client.email for client in mailing.clients.all()],
                fail_silently=False,
            )
            MailingAttempt.objects.create(mailing=mailing, status='successful', owner=mailing.owner)
        except smtplib.SMTPException as e:
            MailingAttempt.objects.create(mailing=mailing, status='failed', response=str(e), owner=mailing.owner)

def check_frequency(last_attempt, frequency):
    interval_mapping = {
        'daily': timezone.timedelta(days=1),
        'weekly': timezone.timedelta(weeks=1),
        'monthly': timezone.timedelta(weeks=4),
    }
    return timezone.now() - last_attempt.timestamp >= interval_mapping.get(frequency, timezone.timedelta(days=1))
